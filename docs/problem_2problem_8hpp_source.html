380  template <typename T>
381  void
382  wrap_function(ObjectiveFunction<T> f, const std::string &name,
383  const common::OptimizationType optimization_type = common::OptimizationType::Minimization,
384  const std::optional<T> lb = std::nullopt, const std::optional<T> ub = std::nullopt,
385  std::optional<VariablesTransformationFunction<T>> transform_variables_function = std::nullopt,
386  std::optional<ObjectiveTransformationFunction> transform_objectives_function = std::nullopt,
387  std::optional<CalculateObjectiveFunction<T>> calculate_objective = std::nullopt)
388  {
389  auto &factory = ProblemFactoryType<Problem<T>>::instance();
390 
391  int id = factory.check_or_get_next_available(1, name);
392 
393  auto constraint = Constraint<T>(1, lb.value_or(std::numeric_limits<T>::lowest()),
394  ub.value_or(std::numeric_limits<T>::max()));
395 
396  auto tx = transform_variables_function.value_or(utils::identity<std::vector<T>, int>);
397  auto ty = transform_objectives_function.value_or(utils::identity<double, int>);
398 
399  factory.include(name, id,
400  [f, name, id, optimization_type, constraint, tx, ty, calculate_objective](const int iid,
401  const int dim) {
402  auto objective = calculate_objective ? calculate_objective.value()(iid, dim)
403  : Solution<T>(dim, optimization_type);
404 
405  return std::make_unique<WrappedProblem<T>>(f, name, dim, id, iid, optimization_type,
406  constraint, tx, ty, objective);
407  });
408  }
