695  void fill_up(size_t i_error, size_t j_evals)
696  {
697  auto &mat = current_eah();
698  const auto ibound = _range_error.size();
699  auto jbound = _range_evals.size();
700 
701  if (_current.has_opt || _current.max_min == common::OptimizationType::Minimization)
702  {
703  for (auto i = i_error; i < ibound; i++)
704  {
705  // If we reach a 1 on first col of this row, no need to continue.
706  // TODO: check this, out of bound error for mscv
707  if (mat[i][std::min(j_evals, jbound - 1)] == 1)
708  {
709  break;
710  }
711  for (auto j = j_evals; j < jbound; j++)
712  {
713  // If we reach a 1 on this col, no need to fill
714  // the remaining columns for the next rows..
715  if (mat[i][j] == 1)
716  {
717  jbound = j;
718  break;
719  }
720  mat[i][j] = true;
721  } // for j
722  } // for i
723  }
724  else
725  {
726  assert(
727  !_current.has_opt && _current.max_min == common::OptimizationType::Maximization);
728  for (auto i = i_error; i >= 1; i--)
729  {
730  // If we reach a 1 on first col of this row, no need to continue.
731  if (mat[i - 1][j_evals] == 1)
732  {
733  continue;
734  }
735  for (auto j = j_evals; j < jbound; j++)
736  {
737  // If we reach a 1 on this col, no need to fill
738  // the remaining columns for the next rows..
739  if (mat[i - 1][j] == 1)
740  {
741  jbound = j;
742  break;
743  }
744  mat[i - 1][j] = true;
745  } // for j
746  } // for i
747  }
