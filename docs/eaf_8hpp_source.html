462  Type operator()(const EAF& logger)
463  {
464  // Not a perfect guard, because this only tests the last attached problem.
465  // But, there is no way to ensure that all problems seen by the loggers
466  // have been of the same type.
467  // This is at least a check.
468  assert(_optim_type == logger.problem().optimization_type.type());
469 
470  // Input:
471  std::vector<eaf::Front> fronts;
472  // Flatten the front in the runs.
473  for(const auto& suite_pb : logger.data()) {
474  for(const auto& pb_dim : suite_pb.second) {
475  for(const auto& dim_ins : pb_dim.second) {
476  for(const auto& ins_runs : dim_ins.second) {
477  for(const auto& run_front : ins_runs.second) {
478  fronts.push_back(run_front.second);
479  } } } } }
480 
481  const size_t nb_runs = fronts.size();
482  IOH_DBG(note, nb_runs << " runs in the EAF logger")
483  assert(nb_runs > 0);
484 
485  // Copy and sort data.
486  std::vector<eaf::RunPoint> data_time, data_qual;
487  size_t total_nb_points = 0;
488  for(size_t run = 0; run < fronts.size(); ++run) {
489  assert(fronts[run].size() > 0);
490  // Cumulative size.
491  total_nb_points += fronts[run].size();
492 
493  // Queues: time = x in papers, qual = y.
494  for(const eaf::Point& p : fronts[run]) {
495  // time, qual, id of the related run
496  data_time.push_back(eaf::RunPoint(p.time, p.qual, run));
497  data_qual.push_back(eaf::RunPoint(p.time, p.qual, run));
498  }
499  }
500  IOH_DBG(note, total_nb_points << " front points in the EAF logger")
501 
502  // Sort the two sets of data.
503  std::sort(std::begin(data_time),std::end(data_time), ascending_time); // Time always ascend.
504 
505  if(_optim_type == common::OptimizationType::Minimization ) {
506  std::sort(std::begin(data_qual),std::end(data_qual), descending_qual);
507  } else {
508  std::sort(std::begin(data_qual),std::end(data_qual), ascending_qual);
509  }
510 
511  // Defaults to all levels asked.
512  if(_attlevels.empty()) {
513  for(size_t r=0; r<nb_runs; ++r) {
514  _attlevels.push_back(r);
515  }
516  }
517  // Output:
518  Type levels;
519 
520  // The algorithm.
521  for(const size_t level : _attlevels) {
522  IOH_DBG(debug, "Parse level " << level)
523  eaf::Front eaf;
524  size_t it = 0; // current time index
525  size_t iq = 0; // current qual index
526  size_t nb_attained = 0;
527 
528  // Can hold negative values.
529  std::vector<long> attained(nb_runs, 0);
530  std::vector<char> attained_runs(nb_runs, 0);
531 
532  // Start at upper-left corner.
533  size_t run = data_time[it].run;
534  assert(run < nb_runs);
535  attained[run]++;
536  nb_attained++;
537 
538  // Result for this level.
539  eaf::Front level_front;
540 
541  do { // while it < total and iq < total
542  IOH_DBG(debug, "> Parse ascending time from " << it << " (time=" << data_time[it].time << ")" )
543 
544  // Until the desired attainment level is reached.
545  while(it < total_nb_points-1 and
546  (nb_attained < level or data_time[it].time == data_time[it+1].time)) {
547  it++;
548  assert(it < total_nb_points);
549  if(is_better_or_eq(data_time[it].qual, data_qual[iq].qual)) {
550  run = data_time[it].run;
551  assert(run < nb_runs);
552  if(attained[run] == 0) {
553  nb_attained++; }
554  attained[run]++;
555  } // it.qual <= iq.qual
556  } // while it < total_nb_points
557  IOH_DBG(debug, "> Reached level at " << it)
558  // IOH_DBG_DUMP(xdebug, attained, "attained_right_{n}.dat");
559 
560  if(nb_attained >= level) {
561  IOH_DBG(debug, ">> Level is to be saved")
562  IOH_DBG(debug, ">> Parse descending qual from " << iq << " (qual=" << data_qual[iq].qual << ")")
563  // Until the desired attainment level is no longer reached.
564  do { // while nb_attained >= level and iq < total
565 
566  for(size_t r=0; r < nb_runs; ++r) {
567  attained_runs[r] = attained[r] >= 1 ? 1 : 0;
568  }
569 
570  do { // while iq < total and iq.qual == iq-1.qual
571  assert(it < total_nb_points);
572  assert(iq < total_nb_points);
573  if( data_qual[iq].time <= data_time[it].time) {
574  run = data_qual[iq].run;
575  assert(run < nb_runs);
576  attained[run]--; // Can be negative.
577  if(attained[run] == 0) {
578  assert(nb_attained > 0);
579  nb_attained--; }
580  } // if iq.time <= it.time
581  // IOH_DBG_DUMP(xdebug, attained, "attained_down_{n}.dat");
582  iq++;
583  assert(iq-1 < total_nb_points);
584  } while(iq < total_nb_points and data_qual[iq].qual == data_qual[iq-1].qual);
585  } while(nb_attained >= level and iq < total_nb_points);
586  IOH_DBG(debug, ">> Reached level at " << iq)
587  // assert(nb_attained < level);
588  assert(it < total_nb_points);
589  assert(iq-1 < total_nb_points);
590  IOH_DBG(debug, ">> Front level point: (" << data_time[it].time << "," << data_qual[iq-1].qual << ")")
591  level_front.push_back( eaf::RunPoint(
592  data_time[it].time,
593  data_qual[iq-1].qual,
594  std::accumulate( std::begin(attained_runs), std::end(attained_runs), 0)
595  ));
596  } // if nb_attained
597 
598  } while(it < total_nb_points-1 and iq < total_nb_points);
599 
600  // Save this level.
601  IOH_DBG(note, "> Level " << level << " has " << level_front.size() << " points")
602  assert(level_front.size() > 0);
603  assert(levels.find(level) == std::end(levels));
604  levels[level] = level_front;
605  #ifndef NDEBUG
606  IOH_DBG(xdebug,">> Involved runs:")
607  IOH_DBG(xdebug,">>> " << level_front[0].run)
608  for(size_t i = 1; i < level_front.size(); ++i) {
609  if(level_front[i].run != level_front[i-1].run) {
610  IOH_DBG(xdebug,">>> " << level_front[i].run)
611  }
612  assert(level_front[i].run == level_front[i-1].run);
613  }
614  #endif
615  } // l in levels
616 
617  IOH_DBG(note, "Ended with " << levels.size() << " levels")
618  assert(levels.size() > 0);
619  assert(levels.size() <= nb_runs);
620  return levels;
621  }

679  eaf::Point extremum(const eaf::Front& front) const
680  {
681  size_t worst_time;
682  double worst_qual;
683  if(_optim_type.type() == common::OptimizationType::Minimization) {
684  worst_time = std::numeric_limits<size_t>::lowest();
685  worst_qual = -std::numeric_limits<double>::infinity();

688  worst_qual = std::numeric_limits<double>::infinity();

691  if(p.time > worst_time) { worst_time = p.time; }
692  if(_optim_type(worst_qual, p.qual)) { worst_qual = p.qual; }
693  }
694  return eaf::Point(worst_qual, worst_time);
695  }
