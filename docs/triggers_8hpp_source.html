264  class OnImprovement : public logger::Trigger
265  {
266  protected:
268  double _best;

272

289 
290  }
291 
293  double best() const { return _best; }
294 
296  common::OptimizationType type() const { return _type.type(); }
297 
299  bool operator()(const logger::Info &log_info, const problem::MetaData &pb_info) override
300  {
301  if (not _has_type)
302  {
303  _type = pb_info.optimization_type;
304  if (_type.type() == common::OptimizationType::Minimization)
305  {
306  IOH_DBG(debug, "reconfigure problem type as minimization")
307  _best = std::numeric_limits<double>::infinity();
308  }
309  else
310  {
311  IOH_DBG(debug, "reconfigure problem type as maximization")
312  _best = -std::numeric_limits<double>::infinity();
313  }
314  _has_type = true;
315  }
316  // We do not use logger::Info::transformed_y_best below,
317  // because all fields of logger::Info are updated before the trigger see them.
318  // That would force to test for equality to trigger on improvement,
319  // and we only want to trigger on strict inequality.
320  assert(_has_type);
321  if (_type(log_info.transformed_y, _best))
322  {
323  _best = log_info.transformed_y;
324  IOH_DBG(debug, "triggered on improvement by " << log_info.transformed_y << " / " << _best)
325  return true;
326  }
327  IOH_DBG(xdebug, "not triggered on improvement by " << log_info.transformed_y << " / " << _best)
328  return false;
329  }
330 
332  void reset() override { _has_type = false; }
333  };
