65  {
66  int epistasis_result;
67  const auto number_of_variables = static_cast<int>(variables.size());
68  std::vector<int> new_variables;
69  new_variables.reserve(number_of_variables);
70  auto h = 0;
71  while (h + v - 1 < number_of_variables)
72  {
73  auto i = 0;
74  while (i < v)
75  {
76  epistasis_result = -1;
77  for (auto j = 0; j < v; ++j)
78  if (v - j - 1 != (v - i - 1 - 1) % 4)
79  epistasis_result = epistasis_result == -1
80  ? variables[static_cast<size_t>(j) + h]
81  : epistasis_result != variables[static_cast<size_t>(j) + h];
82 
83  new_variables.push_back(epistasis_result);
84  ++i;
85  }
86  h += v;
87  }
88  if (number_of_variables - h > 0)
89  {
90  v = static_cast<int>(number_of_variables) - h;
91  for (auto i = 0; i < v; i++)
92  {
93  epistasis_result = -1;
94  for (auto j = 0; j < v; ++j)
95  if (v - j - 1 != (v - i - 1 - 1) % 4)
96  epistasis_result = epistasis_result == -1

98  : epistasis_result != variables.at(static_cast<size_t>(h) + j);
99  new_variables.push_back(epistasis_result);

184  {
185  const auto thresholdFor1 = (mu >> 1) + (mu & 1);
186  int temp;
187  const auto dim = static_cast<int>(xIn.size());
188  const auto temp_dim = dim / mu;
189  if (static_cast<int>(xOut.size()) != temp_dim)
190  {
191  xOut.resize(temp_dim);
192  }
193  auto i = 0;
194  auto j = 0;
195  auto ones = 0;
196  auto flush = mu;
197  while (i < dim && j < temp_dim)
198  {
199  if (xIn[i] == 1)
200  {
201  ones += 1;
202  }
203  i += 1;
204  if (i >= flush)
205  {
206  flush += mu;
207  if (ones >= thresholdFor1)
208  {
209  temp = 1;
210  }
211  else
212  {
213  temp = 0;
214  }
215  xOut[j] = temp;
216  j += 1;
217  ones = 0;
218  }

270  {
271  int i, j, start;
272  std::vector<int> r(q + 1, 0);
273  r[0] = 0;
274  const auto max = max_gamma(q);
275  if (gamma <= 0)
276  {
277  start = 0;
278  }
279  else
280  {
281  start = q - 1 - static_cast<int>(0.5 + sqrt(0.25 + ((max - gamma) << 1)));
282  }
283  auto k = 0;
284  for (j = 1; j <= start; j++)
285  {
286  if ((j & 1) != 0)
287  {
288  r[j] = q - k;
289  }
290  else
291  {
292  k = k + 1;
293  r[j] = k;
294  }
295  }
296  for (; j <= q; j++)
297  {
298  k = k + 1;
299  if ((start & 1) != 0)
300  {
301  r[j] = q - k;
302  }
303  else
304  {
305  r[j] = k;
306  }
307  }
308  const auto upper = gamma - max + ((q - start - 1) * (q - start) >> 1);
309  j--;
310  for (i = 1; i <= upper; i++)
311  {
312  j = j - 1;
313 
314  if (j > 0)
315  {
316  const auto t = r[j];
317  r[j] = r[q];
318  r[q] = t;
319  }
320  }
321 
322  std::vector<int> r2(1 + q, 0);
323  for (i = 0; i <= q; i++)
324  {

330  inline int ruggedness_translate(int gamma, int q)
331  {
332  int j, k;
333 
334  if (gamma <= 0)
335  {
336  return 0;
337  }
338  const auto g = gamma;
339  const auto max = max_gamma(q);
340  const auto lastUpper = (q >> 1) * ((q + 1) >> 1);
341  if (g <= lastUpper)
342  {
343  j = abs(static_cast<int>((q + 2) * 0.5 - sqrt(q * q * 0.25 + 1 - g)));
344 
345  k = g - (q + 2) * j + j * j + q;
346  return k + 1 + (((q + 2) * j - j * j - q - 1) << 1) - (j - 1);
347  }
348 
349  j = abs(static_cast<int>((q % 2 + 1) * 0.5 + sqrt((1 - q % 2) * 0.25 + g - 1 - lastUpper)));
350 
351  k = g - ((j - q % 2) * (j - 1) + 1 + lastUpper);
352 
353  return max - k - (2 * j * j - j) - q % 2 * (-2 * j + 1);
354  }
